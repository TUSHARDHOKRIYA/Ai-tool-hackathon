REVISION 1 — Severity Index from Your Existing ResNet
You do NOT need to retrain your model. Your ResNet already outputs a confidence probability. That number IS the severity score — you just need to reframe how you read it.
Key Insight: Confidence Score = Severity
When your model says 'Bleached: 0.87' — it already knows HOW bleached. The 0.87 is the severity. You were just hiding this information by forcing a yes/no output.

Step 1 — Define Strict Thresholds (All in One Place)
Never scatter magic numbers across your codebase. Every threshold lives in one constants file so scientists can tune them:
# constants.py — single source of truth for all thresholds
THRESHOLDS = {
    "sst": {
        "baseline": 28.0,   # healthy max for most coral species
        "watch":    28.5,   # begin monitoring
        "stress":   29.0,   # thermal stress begins
        "bleach":   29.5,   # NOAA standard bleaching threshold
        "severe":   30.5,   # mass bleaching likely
        "critical": 31.5,   # mortality risk
    },
    "ph": {
        "healthy":  8.3,
        "watch":    8.1,
        "stress":   7.9,
        "critical": 7.75,
    },
    "uv": { "low": 3, "moderate": 6, "high": 8, "extreme": 11 },
    "dhw": {
        "watch":  0,   # begin heat stress accumulation
        "alert":  4,   # bleaching likely (NOAA standard)
        "severe": 8,   # mass bleaching / mortality
    },
    "health_score": {
        "healthy":   75,
        "watch":     55,
        "stressed":  40,
        "bleached":  25,
        "critical":  10,
    },
    "decline_rate": { "mild": 10, "moderate": 20, "rapid": 30 },
    "days_no_data": { "stale": 30, "blind": 60 }
}
Step 2 — Strict Multi-Factor Threat Engine
Every threat requires MULTIPLE corroborating signals. One hot day is not a bleaching event. One bad pH reading is not acidification. The engine cross-checks everything:
SEVERITY_ORDER = {
    "CRITICAL": 0, "HIGH": 1, "MEDIUM": 2, "WATCH": 3, "LOW": 4, "INFO": 5
}
def evaluate_threats(snapshot, history, ocean_data):
    threats = []
    sst     = ocean_data['sst']
    ph      = ocean_data['ph']
    uv      = ocean_data['uv']
    dhw     = ocean_data['dhw']
    score   = snapshot['health_score']
    prev    = history[-1]['health_score'] if history else score
    decline = prev - score  # positive = getting worse
    T = THRESHOLDS
    # ── THERMAL: SST must be confirmed by DHW accumulation ──
    # A single spike without heat buildup is NOT a bleaching event
    if sst >= T['sst']['critical'] and dhw >= T['dhw']['severe']:
        threats.append(threat('THERMAL', 'CRITICAL',
            f'SST {sst}C + DHW {dhw} — mass mortality risk'))
    elif sst >= T['sst']['severe'] and dhw >= T['dhw']['alert']:
        threats.append(threat('THERMAL', 'HIGH',
            f'SST {sst}C sustained — mass bleaching likely'))
    elif sst >= T['sst']['bleach'] and dhw >= T['dhw']['watch']:
        threats.append(threat('THERMAL', 'MEDIUM',
            f'SST {sst}C at bleach threshold, DHW accumulating'))
    elif sst >= T['sst']['stress']:
        threats.append(threat('THERMAL', 'WATCH',
            f'SST {sst}C — stress window, monitor daily'))
    # ── ACIDIFICATION: pH severity escalates if reef already weak ──
    if ph <= T['ph']['critical']:
        sev = 'CRITICAL' if score < T['health_score']['stressed'] else 'HIGH'
        threats.append(threat('ACIDIFICATION', sev,
            f'pH {ph} — carbonate dissolution active, skeleton damage'))
    elif ph <= T['ph']['stress']:
        sev = 'HIGH' if score < T['health_score']['watch'] else 'MEDIUM'
        threats.append(threat('ACIDIFICATION', sev,
            f'pH {ph} — below safe threshold, compounding bleach stress'))
    elif ph <= T['ph']['watch']:
        threats.append(threat('ACIDIFICATION', 'WATCH',
            f'pH {ph} — approaching stress zone'))
    # ── RAPID DECLINE: only meaningful with history to compare ──
    if history and decline >= T['decline_rate']['rapid']:
        threats.append(threat('RAPID_DECLINE', 'CRITICAL',
            f'Score dropped {decline}pts — emergency monitoring needed'))
    elif history and decline >= T['decline_rate']['moderate']:
        threats.append(threat('RAPID_DECLINE', 'HIGH',
            f'Score dropped {decline}pts — accelerating event'))
    elif history and decline >= T['decline_rate']['mild']:
        threats.append(threat('RAPID_DECLINE', 'MEDIUM',
            f'Score dropped {decline}pts — declining trend confirmed'))
    # ── COMPOUND STRESS: multiple stressors = independent threat ──
    # Most dangerous scenario — triggers even if individual ones are WATCH
    active = sum([
        sst   >= T['sst']['stress'],
        ph    <= T['ph']['stress'],
        uv    >= T['uv']['high'],
        dhw   >= T['dhw']['watch'],
        score  < T['health_score']['stressed'],
    ])
    if active >= 4:
        threats.append(threat('COMPOUND', 'CRITICAL',
            f'{active}/5 stressors active — recovery unlikely without intervention'))
    elif active >= 3:
        threats.append(threat('COMPOUND', 'HIGH',
            f'{active}/5 stressors active — compounded risk'))
    # ── UV: only dangerous when combined with thermal stress ──
    if uv >= T['uv']['extreme'] and sst >= T['sst']['stress']:
        threats.append(threat('UV_AMPLIFICATION', 'HIGH',
            f'UV {uv} + SST {sst}C — UV amplifying thermal damage ~30%'))
    elif uv >= T['uv']['high']:
        threats.append(threat('UV', 'LOW',
            f'UV {uv} — elevated but not critical standalone'))
    # ── DATA GAP: unknown status during risk window ──
    days = days_since_last_upload(history)
    if days >= T['days_no_data']['blind']:
        threats.append(threat('NO_DATA', 'HIGH',
            f'{days} days without survey — reef status unknown'))
    elif days >= T['days_no_data']['stale']:
        threats.append(threat('NO_DATA', 'MEDIUM',
            f'{days} days since last upload — data stale'))
    return sorted(threats, key=lambda t: SEVERITY_ORDER[t['severity']])
def threat(type, severity, message):
    return {'type': type, 'severity': severity, 'message': message}

How Clustering Works
Image arrives → GPS extracted → search DB for any reef within 2 km
Match found → link image to that reef's history (same reef_id)
No match → create new reef zone, assign unique reef_id
Multiple matches → link to nearest reef centroid
Researchers can override and manually assign to a specific reef_id
PostGIS Query for Matching
-- Find nearest reef within 2km of uploaded image coordinates
SELECT reef_id, reef_name,
       ST_Distance(location::geography,
         ST_MakePoint(:lon,:lat)::geography) AS dist_meters
FROM reefs
WHERE ST_DWithin(location::geography,
        ST_MakePoint(:lon,:lat)::geography, 2000)
ORDER BY dist_meters ASC
LIMIT 1;

Reef Database Schema
-- Run this in Supabase SQL Editor
CREATE EXTENSION IF NOT EXISTS postgis;
CREATE TABLE reefs (
  reef_id    UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  reef_name  TEXT,
  location   GEOGRAPHY(Point, 4326),   -- PostGIS geo point
  created_at TIMESTAMPTZ DEFAULT now()
);
CREATE TABLE reef_snapshots (
  id            UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  reef_id       UUID REFERENCES reefs(reef_id),
  captured_at   TIMESTAMPTZ,            -- from EXIF timestamp
  uploaded_at   TIMESTAMPTZ DEFAULT now(),
  health_score  INTEGER,                -- 0-100
  bleach_stage  TEXT,                   -- Healthy / Stress / Partial etc.
  bleach_confidence FLOAT,              -- raw model probability
  sst_celsius   FLOAT,                  -- from NOAA API
  ocean_ph      FLOAT,                  -- from pH buoy API
  uv_index      FLOAT,                  -- from OpenUV API
  threats       JSONB,                  -- array of threat objects
  image_url     TEXT,                   -- Cloudinary / S3 URL
  uploaded_by   TEXT
);
-- Index for fast geo queries
CREATE INDEX reefs_location_idx ON reefs USING GIST(location);

DHW — The Most Important Metric
Degree Heating Weeks (DHW) is the gold standard used by NOAA Coral Reef Watch. It measures how long a reef has been exposed to heat stress. No API key needed — NOAA provides it free:
DHW 0: No stress
DHW 4: Bleaching likely
DHW 8: Severe bleaching / mortality
NOAA Coral Reef Watch API: coralreefwatch.noaa.gov/product/5km — no key required
