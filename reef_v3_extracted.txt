Every threshold lives in one constants file so scientists can tune them:# constants.py — single source of truth for all thresholdsTHRESHOLDS = {    &quot;sst&quot;: {        &quot;baseline&quot;: 28.0,   # healthy max for most coral species        &quot;watch&quot;:    28.5,   # begin monitoring        &quot;stress&quot;:   29.0,   # thermal stress begins        &quot;bleach&quot;:   29.5,   # NOAA standard bleaching threshold        &quot;severe&quot;:   30.5,   # mass bleaching likely        &quot;critical&quot;: 31.5,   # mortality risk    },    &quot;ph&quot;: {        &quot;healthy&quot;:  8.3,        &quot;watch&quot;:    8.1,        &quot;stress&quot;:   7.9,        &quot;critical&quot;: 7.75,    },    &quot;uv&quot;: { &quot;low&quot;: 3, &quot;moderate&quot;: 6, &quot;high&quot;: 8, &quot;extreme&quot;: 11 },    &quot;dhw&quot;: {        &quot;watch&quot;:  0,   # begin heat stress accumulation        &quot;alert&quot;:  4,   # bleaching likely (NOAA standard)        &quot;severe&quot;: 8,   # mass bleaching / mortality    },    &quot;health_score&quot;: {        &quot;healthy&quot;:   75,        &quot;watch&quot;:     55,        &quot;stressed&quot;:  40,        &quot;bleached&quot;:  25,        &quot;critical&quot;:  10,    },    &quot;decline_rate&quot;: { &quot;mild&quot;: 10, &quot;moderate&quot;: 20, &quot;rapid&quot;: 30 },    &quot;days_no_data&quot;: { &quot;stale&quot;: 30, &quot;blind&quot;: 60 }}Step 2 — Strict Multi-Factor Threat EngineEvery threat requires MULTIPLE corroborating signals. One hot day is not a bleaching event. One bad pH reading is not acidification. The engine cross-checks everything:SEVERITY_ORDER = {    &quot;CRITICAL&quot;: 0, &quot;HIGH&quot;: 1, &quot;MEDIUM&quot;: 2, &quot;WATCH&quot;: 3, &quot;LOW&quot;: 4, &quot;INFO&quot;: 5}def evaluate_threats(snapshot, history, ocean_data):    threats = []    sst     = ocean_data[&apos;sst&apos;]    ph      = ocean_data[&apos;ph&apos;]    uv      = ocean_data[&apos;uv&apos;]    dhw     = ocean_data[&apos;dhw&apos;]    score   = snapshot[&apos;health_score&apos;]    prev    = history[-1][&apos;health_score&apos;] if history else score    decline = prev - score  # positive = getting worse    T = THRESHOLDS    # ── THERMAL: SST must be confirmed by DHW accumulation ──    # A single spike without heat buildup is NOT a bleaching event    if sst >= T[&apos;sst&apos;][&apos;critical&apos;] and dhw >= T[&apos;dhw&apos;][&apos;severe&apos;]:        threats.append(threat(&apos;THERMAL&apos;, &apos;CRITICAL&apos;,            f&apos;SST {sst}C + DHW {dhw} — mass mortality risk&apos;))    elif sst >= T[&apos;sst&apos;][&apos;severe&apos;] and dhw >= T[&apos;dhw&apos;][&apos;alert&apos;]:        threats.append(threat(&apos;THERMAL&apos;, &apos;HIGH&apos;,            f&apos;SST {sst}C sustained — mass bleaching likely&apos;))    elif sst >= T[&apos;sst&apos;][&apos;bleach&apos;] and dhw >= T[&apos;dhw&apos;][&apos;watch&apos;]:        threats.append(threat(&apos;THERMAL&apos;, &apos;MEDIUM&apos;,            f&apos;SST {sst}C at bleach threshold, DHW accumulating&apos;))    elif sst >= T[&apos;sst&apos;][&apos;stress&apos;]:        threats.append(threat(&apos;THERMAL&apos;, &apos;WATCH&apos;,            f&apos;SST {sst}C — stress window, monitor daily&apos;))    # ── ACIDIFICATION: pH severity escalates if reef already weak ──    if ph <= T[&apos;ph&apos;][&apos;critical&apos;]:        sev = &apos;CRITICAL&apos; if score < T[&apos;health_score&apos;][&apos;stressed&apos;] else &apos;HIGH&apos;        threats.append(threat(&apos;ACIDIFICATION&apos;, sev,            f&apos;pH {ph} — carbonate dissolution active, skeleton damage&apos;))    elif ph <= T[&apos;ph&apos;][&apos;stress&apos;]:        sev = &apos;HIGH&apos; if score < T[&apos;health_score&apos;][&apos;watch&apos;] else &apos;MEDIUM&apos;        threats.append(threat(&apos;ACIDIFICATION&apos;, sev,            f&apos;pH {ph} — below safe threshold, compounding bleach stress&apos;))    elif ph <= T[&apos;ph&apos;][&apos;watch&apos;]:        threats.append(threat(&apos;ACIDIFICATION&apos;, &apos;WATCH&apos;,            f&apos;pH {ph} — approaching stress zone&apos;))    # ── RAPID DECLINE: only meaningful with history to compare ──    if history and decline >= T[&apos;decline_rate&apos;][&apos;rapid&apos;]:        threats.append(threat(&apos;RAPID_DECLINE&apos;, &apos;CRITICAL&apos;,            f&apos;Score dropped {decline}pts — emergency monitoring needed&apos;))    elif history and decline >= T[&apos;decline_rate&apos;][&apos;moderate&apos;]:        threats.append(threat(&apos;RAPID_DECLINE&apos;, &apos;HIGH&apos;,            f&apos;Score dropped {decline}pts — accelerating event&apos;))    elif history and decline >= T[&apos;decline_rate&apos;][&apos;mild&apos;]:        threats.append(threat(&apos;RAPID_DECLINE&apos;, &apos;MEDIUM&apos;,            f&apos;Score dropped {decline}pts — declining trend confirmed&apos;))    # ── COMPOUND STRESS: multiple stressors = independent threat ──    # Most dangerous scenario — triggers even if individual ones are WATCH    active = sum([        sst   >= T[&apos;sst&apos;][&apos;stress&apos;],        ph    <= T[&apos;ph&apos;][&apos;stress&apos;],        uv    >= T[&apos;uv&apos;][&apos;high&apos;],        dhw   >= T[&apos;dhw&apos;][&apos;watch&apos;],        score  < T[&apos;health_score&apos;][&apos;stressed&apos;],    ])    if active >= 4:        threats.append(threat(&apos;COMPOUND&apos;, &apos;CRITICAL&apos;,            f&apos;{active}/5 stressors active — recovery unlikely without intervention&apos;))    elif active >= 3:        threats.append(threat(&apos;COMPOUND&apos;, &apos;HIGH&apos;,            f&apos;{active}/5 stressors active — compounded risk&apos;))    # ── UV: only dangerous when combined with thermal stress ──    if uv >= T[&apos;uv&apos;][&apos;extreme&apos;] and sst >= T[&apos;sst&apos;][&apos;stress&apos;]:        threats.append(threat(&apos;UV_AMPLIFICATION&apos;, &apos;HIGH&apos;,            f&apos;UV {uv} + SST {sst}C — UV amplifying thermal damage ~30%&apos;))    elif uv >= T[&apos;uv&apos;][&apos;high&apos;]:        threats.append(threat(&apos;UV&apos;, &apos;LOW&apos;,            f&apos;UV {uv} — elevated but not critical standalone&apos;))    # ── DATA GAP: unknown status during risk window ──    days = days_since_last_upload(history)    if days >= T[&apos;days_no_data&apos;][&apos;blind&apos;]:        threats.append(threat(&apos;NO_DATA&apos;, &apos;HIGH&apos;,            f&apos;{days} days without survey — reef status unknown&apos;))    elif days >= T[&apos;days_no_data&apos;][&apos;stale&apos;]:        threats.append(threat(&apos;NO_DATA&apos;, &apos;MEDIUM&apos;,            f&apos;{days} days since last upload — data stale&apos;))    return sorted(threats, key=lambda t: SEVERITY_ORDER[t[&apos;severity&apos;]])def threat(type, severity, message):    return {&apos;type&apos;: type, &apos;severity&apos;: severity, &apos;message&apos;: message}REVISION 2 — Region / Cluster Radius50 meters is too small for ocean scale.
